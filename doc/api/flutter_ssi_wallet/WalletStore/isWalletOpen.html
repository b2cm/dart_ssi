bool isWalletOpen() {
  if (_keyBox == null ||
      _issuingHistory == null ||
      _credentialBox == null ||
      _configBox == null ||
      _connection == null)
    return false;
  else
    return (this._keyBox!.isOpen) &&
        (_issuingHistory!.isOpen) &&
        (this._credentialBox!.isOpen) &&
        (this._configBox!.isOpen) &&
        (this._connection!.isOpen);
}
